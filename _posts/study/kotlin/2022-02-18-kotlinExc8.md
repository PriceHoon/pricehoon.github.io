**3-1. 상속과 오버라이딩**
~~~java
//java와 내용적인 개념은 같다.
fun main(){
  //Parent()
  Child().disease()
}

oepn class Parent(){

  open fun doing(){//메서드의 오버라이딩을 위해서 fun앞에도 open을 붙여줘야 한다.
    println("자식을 돌봅니다.")
  }

  open fun disease(){
    println("비염이 있습니다.")
  }
}

class Child():Parent(){
  init{
    println("자식 입니다.")
  }

  override fun disease(){
    super.disease()//부모클레스의 메서드 호출
    println("비염이 없습니다.")
  }
}
~~~
